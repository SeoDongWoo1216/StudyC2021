## 퀴즈
: 두 값을 입력받고, 두 값의 사칙연산을 수행해서 연산 결과를 출력하는 프로그램
```C
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main() 
{

	int firNum = 0;
	int secNum = 0;
	char cal = '+';

	printf("----------------------------------\n");
	printf("Calculator_v2\n");
	printf("Created : 2021-04-28\n");
	printf("Author : 서동우\n");
	printf("----------------------------------\n");
	printf("두 정수를 입력하세요 : ");
	scanf("%d %d", &firNum, &secNum);
	printf("연산기호를 입력하세요(+,-,*,/) : ");
	scanf(" %c", &cal);
	printf("----------------------------------\n");

	switch (cal)
	{
	case '+':
		printf("%d + %d = %d\n", firNum, secNum, firNum + secNum);
		break;

	case '-':
		printf("%d - %d = %d\n", firNum, secNum, firNum - secNum);
		break;

	case '*':
		printf("%d * %d = %d\n", firNum, secNum, firNum * secNum);
		break;

	case '/':
		if (secNum == 0)
		{
			printf("0으로는 나눌 수 없습니다");
		}
		else
		{
			printf("%d / %d = %.2lf\n", firNum, secNum, (double)firNum / (double)secNum);
			printf("%d %% %d = %d\n", firNum, secNum, firNum % secNum);
		}
		break;

	default:
		printf("연산기호를 입력해주세요");
		break;
	}
	printf("\n");
	return 0;
}
```

### 무한반복 계산기 
```C
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main() {

	int firNum = 0;
	int secNum = 0;
	char cal = '+';
	int repeatNum = 1;

	while (repeatNum)
	{
		printf("----------------------------------\n");
		printf("Calculator_v2\n");
		printf("Created : 2021-04-28\n");
		printf("Author : 서동우\n");
		printf("----------------------------------\n");
		printf("두 정수를 입력하세요 : ");
		scanf("%d %d", &firNum, &secNum);
		printf("연산기호를 입력하세요(+,-,*,/) : ");
		scanf(" %c", &cal);
		printf("----------------------------------\n");

		switch (cal)
		{
		case '+':
			printf("%d + %d = %d\n", firNum, secNum, firNum + secNum);
			break;

		case '-':
			printf("%d - %d = %d\n", firNum, secNum, firNum - secNum);
			break;

		case '*':
			printf("%d * %d = %d\n", firNum, secNum, firNum * secNum);
			break;

		case '/':
			if (secNum == 0)
			{
				printf("0으로는 나눌 수 없습니다");
			}
			else
			{
				printf("%d / %d = %.2lf\n", firNum, secNum, (double)firNum / (double)secNum);
				printf("%d %% %d = %d\n", firNum, secNum, firNum % secNum);
			}
			break;

		default:
			printf("연산기호를 입력해주세요");
			break;
		}
		printf("\n");
		printf("\n");

		int inputNum;
		printf("반복하시겠습니까? (1:반복, 0:종료) : ");
		scanf("%d", &inputNum);
		if (inputNum == 1)
		{
			repeatNum == 1;
		}
		else if (inputNum == 0)
		{
			break; // exit(1) 로도 종료가능
		}
	}

	return 0;
}
```
